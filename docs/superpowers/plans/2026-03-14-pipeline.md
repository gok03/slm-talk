# Multi-Agent SLM Pipeline Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `/pipeline` page to the existing FastAPI demo runner that orchestrates three SLMs (OCR → JSON → Browser) via A2A protocol and LangGraph, running fully offline.

**Architecture:** A `pipeline/` package added to `demo_runner/` provides A2A types, worker functions, and a LangGraph supervisor. All pipeline state flows through `pipeline_queues` (separate from the existing `run_queues` used by demos 1–5). The browser worker spawns a Playwright MCP subprocess and drives it with Qwen3.5:4b via Ollama's `/api/chat` tool-calling endpoint.

**Tech Stack:** FastAPI, LangGraph 0.2.x, httpx (module-level client), asyncio, MLX-VLM (OCR), Ollama (NuExtract-2.0 + Qwen3.5:4b), Playwright MCP via npx stdio JSON-RPC 2.0.

```python
"""Minimal A2A (Agent-to-Agent) protocol types and in-memory task store."""
from __future__ import annotations

import uuid
from dataclasses import dataclass, field
from typing import Literal

TaskState = Literal["submitted", "working", "completed", "failed"]


@dataclass
class Task:
    id: str
    state: TaskState
    input: dict
    output: dict | None = None
    error: str | None = None


@dataclass
class AgentCard:
    name: str
    description: str
    url: str
    capabilities: list[str] = field(default_factory=list)


# One TaskStore per worker name, keyed by task_id
# Cleared at the start of each pipeline run
task_stores: dict[str, dict[str, Task]] = {
    "ocr": {},
    "json": {},
    "browser": {},
}


def create_task(worker: str, input_data: dict) -> Task:
    task = Task(id=str(uuid.uuid4()), state="submitted", input=input_data)
    task_stores[worker][task.id] = task
    return task


def get_task(worker: str, task_id: str) -> Task | None:
    return task_stores[worker].get(task_id)


def clear_all() -> None:
    for store in task_stores.values():
        store.clear()
```

- [ ] **Step 1: Create `demo_runner/pipeline/workers.py` with the OCR worker**

Use the sub-queue pattern: `_run_mlx_vlm`/`_run_ollama` push plain `{"type":"token","text":"..."}` events without a `step` field. Each worker registers a temporary sub-queue in `pipeline_queues`, calls the helper with that sub-queue's key, and a relay task re-emits every token to `pipeline_queues[run_id]` with the correct `step` field added.

Create `demo_runner/pipeline/workers.py` with:

```python
async def run_ocr_worker(task_id: str, run_id: str, filename: str) -> None:
    """OCR worker: PDF → Markdown via PaddleOCR-VL-1.5."""
    from pipeline.a2a import get_task
    a = _app()

    task = get_task("ocr", task_id)
    if not task:
        return

    task.state = "working"
    t_start = time.perf_counter()
    dest_q = a.pipeline_queues[run_id]

    await dest_q.put({"type": "token", "step": "ocr", "text": "← task received (A2A)\n"})
    await dest_q.put({"type": "token", "step": "ocr", "text": "   state: submitted→working\n"})

    # Resolve file, convert PDF → run_id-scoped PNG
    import base64
    assets_dir = a.ASSETS_DIR
    file_path = assets_dir / filename
    image_path: str | None = None

    if file_path.exists():
        if file_path.suffix.lower() == ".pdf":
            loop = asyncio.get_event_loop()
            png_b64 = await loop.run_in_executor(None, a._pdf_to_base64, file_path)
            tmp = assets_dir / f"_tmp_{run_id}_{file_path.stem}.png"
            tmp.write_bytes(base64.b64decode(png_b64))
            image_path = str(tmp)
            await dest_q.put({"type": "token", "step": "ocr", "text": "   pdf→png · 2× zoom\n"})
        else:
            image_path = str(file_path)

    if not image_path:
        task.state = "failed"
        task.error = "File not found"
        await dest_q.put({"type": "token", "step": "ocr", "text": "   ERROR: file not found\n"})
        return

    # Sub-queue: _run_mlx_vlm pushes plain tokens here; we re-emit with step="ocr"
    sub_run_id = f"_ocr_{run_id}"
    sub_q: asyncio.Queue = asyncio.Queue(maxsize=1024)
    a.pipeline_queues[sub_run_id] = sub_q

    collected: list[str] = []

    async def relay_tokens():
        while True:
            item = await sub_q.get()
            if item["type"] == "token":
                collected.append(item.get("text", ""))
                await dest_q.put({"type": "token", "step": "ocr", "text": item.get("text", "")})
            elif item["type"] in ("done", "error"):
                break

    relay_task = asyncio.create_task(relay_tokens())

    try:
        model_name = "mlx-community/PaddleOCR-VL-1.5-4bit"
        await a._load_mlx_vlm_model(model_name)
        loop = asyncio.get_event_loop()
        await loop.run_in_executor(
            None,
            a._mlx_vlm_stream_blocking,
            sub_run_id,
            model_name,
            "OCR:",
            image_path,
            loop,
            a.pipeline_queues,  # queue_store = pipeline_queues, keyed by sub_run_id
        )
        await relay_task
    except Exception as exc:
        task.state = "failed"
        task.error = str(exc)
        relay_task.cancel()
        await dest_q.put({"type": "token", "step": "ocr", "text": f"   ERROR: {exc}\n"})
        return
    finally:
        a.pipeline_queues.pop(sub_run_id, None)

    markdown = "".join(collected)
    task.output = {"markdown": markdown}
    task.state = "completed"

    duration = round(time.perf_counter() - t_start, 1)
    await dest_q.put({"type": "token", "step": "ocr", "text": "   state: working→completed\n"})
    await dest_q.put({"type": "token", "step": "ocr", "text": "→ result sent (A2A) ✓\n"})
    await dest_q.put({"type": "step_done", "step": "ocr", "duration_s": duration})
```

```python
        a.pipeline_queues.pop(sub_run_id, None)

    raw_json = "".join(collected).strip()

    # Validate JSON
    try:

    task.output = {"json": invoice_dict}
    task.state = "completed"
```

```python
# ── MCP stdio helpers ────────────────────────────────────────────────────────

_mcp_counter = 0


async def _mcp_call(proc: asyncio.subprocess.Process, method: str, params: dict | None = None) -> dict:
    """Send a JSON-RPC 2.0 request to an MCP subprocess and return the result.

    Loops over stdout lines until it finds a response matching the request id.
    This handles Node.js startup noise (log lines, warnings) before responses.
    """
    global _mcp_counter
    _mcp_counter += 1
    req_id = _mcp_counter

    msg = {"jsonrpc": "2.0", "id": req_id, "method": method, "params": params or {}}
    line_bytes = (json.dumps(msg) + "\n").encode()
    proc.stdin.write(line_bytes)
    await proc.stdin.drain()

    while True:
        raw = await proc.stdout.readline()
        if not raw:
            raise EOFError("MCP process closed unexpectedly")
        try:
            data = json.loads(raw)
        except json.JSONDecodeError:
            continue  # skip log lines
        if data.get("id") == req_id:
            if "error" in data:
                raise RuntimeError(f"MCP error [{method}]: {data['error']}")
            return data.get("result") or {}


async def run_browser_worker(task_id: str, run_id: str, invoice_json: dict, form_url: str) -> None:
    """Browser worker: fills /form using Playwright MCP + Qwen3.5:4b tool-calling."""
    from pipeline.a2a import get_task
    a = _app()

    task = get_task("browser", task_id)
    if not task:
        return

    task.state = "working"
    t_start = time.perf_counter()
    dest_q = a.pipeline_queues[run_id]

    await dest_q.put({"type": "token", "step": "browser", "text": "← task received (A2A)\n"})
    await dest_q.put({"type": "token", "step": "browser", "text": "   state: submitted→working\n"})
    await dest_q.put({"type": "token", "step": "browser", "text": f"   spawning Playwright MCP v{PLAYWRIGHT_MCP_VERSION}\n"})

    # Spawn MCP subprocess with --headless flag (standard Playwright CLI flag)
    proc = await asyncio.create_subprocess_exec(
        "npx", f"@playwright/mcp@{PLAYWRIGHT_MCP_VERSION}", "--headless",
        stdin=asyncio.subprocess.PIPE,
        stdout=asyncio.subprocess.PIPE,
        stderr=asyncio.subprocess.DEVNULL,  # suppress Node stderr noise
    )

    submitted = False

    try:
        # 1. MCP initialize
        await _mcp_call(proc, "initialize", {
            "protocolVersion": "2024-11-05",
            "clientInfo": {"name": "browser-worker", "version": "1.0"},
            "capabilities": {},
        })

        # 2. Required: send notifications/initialized (fire-and-forget, no id)
        notif = json.dumps({"jsonrpc": "2.0", "method": "notifications/initialized"}) + "\n"
        proc.stdin.write(notif.encode())
        await proc.stdin.drain()

        await dest_q.put({"type": "token", "step": "browser", "text": "   MCP initialized ✓\n"})

        # 3. List tools
        tools_result = await _mcp_call(proc, "tools/list")
        playwright_tools = tools_result.get("tools", [])

        # 4. Convert to Ollama tool format
        ollama_tools = [
            {
                "type": "function",
                "function": {
                    "name": t["name"],
                    "description": t.get("description", ""),
                    "parameters": t.get("inputSchema", {"type": "object", "properties": {}}),
                },
            }
            for t in playwright_tools
        ]

        await dest_q.put(
            {"type": "token", "step": "browser",
             "text": f"   {len(ollama_tools)} Playwright tools loaded\n"}
        )

        # 5. Tool-calling loop via Ollama /api/chat
        messages = [
            {
                "role": "system",
                "content": (
                    "You are a browser automation agent. "
                    "Use the provided tools to fill out the invoice form at the given URL and submit it. "
                    "Fill every visible field. When done, click the submit button."
                ),
            },
            {
                "role": "user",
                "content": (
                    f"Fill the form at {form_url} with this invoice data: "
                    f"{json.dumps(invoice_json, indent=2)}. "
                    "Fill all fields and submit."
                ),
            },
        ]

        max_iterations = 20
        for iteration in range(max_iterations):
            resp = await a.http_client.post(
                "http://localhost:11434/api/chat",
                json={
                    "model": "qwen3.5:4b",
                    "messages": messages,
                    "tools": ollama_tools,
                    "stream": False,
                },
                timeout=60.0,
            )
            resp.raise_for_status()
            msg = resp.json()["message"]
            messages.append(msg)

            tool_calls = msg.get("tool_calls") or []
            if not tool_calls:
                break  # model done

            for call in tool_calls:
                fn = call.get("function", {})
                tool_name = fn.get("name", "")
                tool_args = fn.get("arguments", {})
                if isinstance(tool_args, str):
                    try:
                        tool_args = json.loads(tool_args)
                    except json.JSONDecodeError:
                        tool_args = {}

                await dest_q.put(
                    {"type": "token", "step": "browser", "text": f"{tool_name} ✓\n"}
                )

                result = await _mcp_call(proc, "tools/call", {
                    "name": tool_name,
                    "arguments": tool_args,
                })
                messages.append({
                    "role": "tool",
                    "content": json.dumps(result),
                    "name": tool_name,
                })

        # 6. Final snapshot: check for submission banner
        try:
            snapshot = await _mcp_call(proc, "tools/call", {
                "name": "browser_snapshot",
                "arguments": {},
            })
            submitted = (
                "submit-banner" in str(snapshot)
                or "Submitted" in str(snapshot)
                or "✓" in str(snapshot)
            )

        task.error = str(exc)
        await dest_q.put({"type": "token", "step": "browser", "text": f"   ERROR: {exc}\n"})
        return
    finally:
        try:
            proc.terminate()
            await asyncio.wait_for(proc.wait(), timeout=5.0)
        except Exception:
            pass

    task.output = {"submitted": submitted}
    task.state = "completed"

    duration = round(time.perf_counter() - t_start, 1)
    status = "→ form submitted (A2A) ✓" if submitted else "→ form submitted (unconfirmed)"
    await dest_q.put({"type": "token", "step": "browser", "text": f"{status}\n"})
    await dest_q.put({"type": "step_done", "step": "browser", "duration_s": duration})
```

```python
"""LangGraph supervisor graph: OCR → JSON → Browser."""
from __future__ import annotations

import asyncio
import time
import uuid
from typing import TypedDict

from langgraph.graph import END, START, StateGraph


class PipelineState(TypedDict):
    pdf_file: str
    run_id: str
    raw_text: str
    invoice_json: dict
    form_submitted: bool
    error: str | None


def _app():
    import app as _a
    return _a


async def _post_task(worker: str, input_data: dict) -> str:
    """POST an A2A task to a worker route; returns task_id."""
    a = _app()
    resp = await a.http_client.post(
        f"http://localhost:8000/workers/{worker}/tasks",
        json=input_data,
        timeout=10.0,
    )
    resp.raise_for_status()
    return resp.json()["task_id"]


async def _poll_task(worker: str, task_id: str, timeout_s: float = 120.0) -> dict:
    """Poll task status until completed or failed. Returns output dict."""
    a = _app()
    deadline = time.monotonic() + timeout_s
    while time.monotonic() < deadline:
        resp = await a.http_client.get(
            f"http://localhost:8000/workers/{worker}/tasks/{task_id}",
            timeout=5.0,
        )
        resp.raise_for_status()
        task_data = resp.json()
        state = task_data.get("state")
        if state == "completed":
            return task_data.get("output") or {}
        if state == "failed":
            raise RuntimeError(task_data.get("error", "worker failed"))
        await asyncio.sleep(0.5)
    raise TimeoutError(f"{worker} worker timed out after {timeout_s}s")


async def ocr_node(state: PipelineState) -> dict:
    a = _app()
    run_id = state["run_id"]
    q = a.pipeline_queues[run_id]

    await q.put({"type": "step_start", "step": "ocr", "label": "OCR Worker — PaddleOCR-VL"})
    await q.put({"type": "token", "step": "supervisor", "text": "→ ocr_node\n"})
    await q.put({"type": "token", "step": "supervisor", "text": "  POST /workers/ocr/tasks\n"})

    t_start = time.perf_counter()
    try:
        task_id = await _post_task("ocr", {"file": state["pdf_file"], "run_id": run_id})
        await q.put({"type": "token", "step": "supervisor", "text": "  poll: working…\n"})
        output = await _poll_task("ocr", task_id)
        duration = round(time.perf_counter() - t_start, 1)
        await q.put({"type": "token", "step": "supervisor", "text": f"  completed ✓ ({duration}s)\n"})
        return {"raw_text": output.get("markdown", ""), "error": None}
    except Exception as exc:
        await q.put({"type": "token", "step": "supervisor", "text": f"  FAILED: {exc}\n"})
        return {"error": str(exc)}


async def json_node(state: PipelineState) -> dict:
    a = _app()
    run_id = state["run_id"]
    q = a.pipeline_queues[run_id]

    if state.get("error"):
        return {}

    await q.put({"type": "step_start", "step": "json", "label": "JSON Worker — NuExtract-2.0"})

async def browser_node(state: PipelineState) -> dict:

    q = a.pipeline_queues[run_id]

    if state.get("error"):

    await q.put({"type": "step_start", "step": "browser", "label": "Browser Worker — Qwen3.5:4b + Playwright MCP"})
    await q.put({"type": "token", "step": "supervisor", "text": "→ browser_node\n"})
    await q.put({"type": "token", "step": "supervisor", "text": "  POST /workers/browser/tasks\n"})

    t_start = time.perf_counter()
    form_url = "http://localhost:8000/form"
    try:
        task_id = await _post_task("browser", {
            "invoice_json": state["invoice_json"],
            "form_url": form_url,
            "run_id": run_id,
        })
        await q.put({"type": "token", "step": "supervisor", "text": "  poll: working…\n"})
        output = await _poll_task("browser", task_id, timeout_s=180.0)
        duration = round(time.perf_counter() - t_start, 1)
        await q.put({"type": "token", "step": "supervisor", "text": f"  completed ✓ ({duration}s)\n"})
        return {"form_submitted": output.get("submitted", False), "error": None}
    except Exception as exc:
        await q.put({"type": "token", "step": "supervisor", "text": f"  FAILED: {exc}\n"})
        return {"error": str(exc)}


def _should_continue(state: PipelineState) -> str:
    return END if state.get("error") else "continue"


def build_graph():
    g = StateGraph(PipelineState)
    g.add_node("ocr_node", ocr_node)
    g.add_node("json_node", json_node)
    g.add_node("browser_node", browser_node)
    g.add_edge(START, "ocr_node")
    g.add_edge("ocr_node", "json_node")
    g.add_edge("json_node", "browser_node")
    g.add_edge("browser_node", END)
    return g.compile()


async def run_pipeline(run_id: str, pdf_file: str) -> None:
    """Entry point called as asyncio.create_task()."""
    a = _app()
    q = a.pipeline_queues[run_id]
    t_start = time.perf_counter()

    # Clear task stores for a fresh run
    from pipeline.a2a import clear_all
    clear_all()

    await q.put({"type": "token", "step": "supervisor", "text": "graph.start()\n"})

    graph = build_graph()
    initial_state: PipelineState = {
        "pdf_file": pdf_file,
        "run_id": run_id,
        "raw_text": "",
        "invoice_json": {},
        "form_submitted": False,
        "error": None,
    }

    try:
        final_state = await graph.ainvoke(initial_state)
    except Exception as exc:
        await q.put({"type": "token", "step": "supervisor", "text": f"graph.error: {exc}\n"})
        final_state = {"error": str(exc), "form_submitted": False}

    total_s = round(time.perf_counter() - t_start, 1)
    await q.put({"type": "token", "step": "supervisor", "text": "graph.end()\n"})
    await q.put({
        "type": "done",
        "metrics": {
            "total_s": total_s,
            "error": final_state.get("error"),
        }
    })
```

```html
<script>
  let evtSource = null;
  let runId = null;
  let stepTimes = {};
  let pipelineStart = null;

  // ── Load file list ──
  async function loadFiles() {
    const resp = await fetch('/api/files');
    const data = await resp.json();
    const sel = document.getElementById('file-select');
    sel.innerHTML = data.files.map(f => `<option value="${f}">${f}</option>`).join('');
  }
  loadFiles();

  // ── Run ──
  async function runPipeline() {
    const file = document.getElementById('file-select').value;
    if (!file) return alert('Select a file first');
    clearAll();
    setGlobalStatus('running');
    document.getElementById('btn-run').disabled = true;
    document.getElementById('btn-stop').disabled = false;
    document.getElementById('run-status').textContent = 'Running…';
    pipelineStart = Date.now();

    const resp = await fetch('/pipeline/run', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({file}),
    });
    const data = await resp.json();
    runId = data.run_id;

    evtSource = new EventSource(`/pipeline/stream/${runId}`);
    evtSource.onmessage = (e) => handleEvent(JSON.parse(e.data));
    evtSource.onerror = () => {
      setGlobalStatus('error');
      document.getElementById('run-status').textContent = 'Stream error';
      finish();
    };
  }

  // ── Stop ──
  async function stopPipeline() {
    if (!runId) return;
    await fetch(`/pipeline/cancel/${runId}`, {method: 'POST'});
    setGlobalStatus('idle');
    document.getElementById('run-status').textContent = 'Stopped';
    finish();
  }

  // ── Event handler ──
  function handleEvent(evt) {
    const type = evt.type;
    const step = evt.step || 'supervisor';

    if (type === 'step_start') {
      stepTimes[step] = Date.now();
      setColStatus(step, 'running');
      appendLog(step, `← ${evt.label || step} started\n`, 'proto');
      return;
    }

    if (type === 'token') {
      const text = evt.text || '';
      const cls = text.startsWith('→') || text.includes('✓') ? 'ok'
                : text.startsWith('   ERROR') ? 'err'
                : text.startsWith('   ') || text.startsWith('←') ? 'proto'
                : 'token';
      appendLog(step, text, cls);
      return;
    }

    if (type === 'step_done') {
      const dur = evt.duration_s || ((Date.now() - (stepTimes[step] || Date.now())) / 1000).toFixed(1);
      setColStatus(step, 'done', `✓ ${dur}s`);
      setColFooter(step, `done — ${dur}s`);
      // Reload iframe when browser step completes
      if (step === 'browser') {
        setTimeout(() => {
          const frame = document.getElementById('form-frame');
          if (frame) frame.contentWindow.location.reload();
        }, 500);
      }
      return;
    }

    if (type === 'step_error') {
      setColStatus(step, 'error', '✗ failed');
      appendLog(step, `ERROR: ${evt.message || ''}\n`, 'err');
      return;
    }

    if (type === 'done') {
      const total = ((Date.now() - pipelineStart) / 1000).toFixed(1);
      setGlobalStatus('done');
      document.getElementById('run-status').textContent = `Done in ${total}s`;
      document.getElementById('metrics-left').textContent =
        `Total: ${total}s`;
      evtSource && evtSource.close();
      finish();
      return;
    }

    if (type === 'error' || type === 'cancelled') {
      setGlobalStatus('error');
      document.getElementById('run-status').textContent = type === 'cancelled' ? 'Cancelled' : `Error: ${evt.message}`;
      evtSource && evtSource.close();
      finish();
    }
  }

  function finish() {
    document.getElementById('btn-run').disabled = false;
    document.getElementById('btn-stop').disabled = true;
    evtSource = null;
    runId = null;
  }

  // ── DOM helpers ──
  function appendLog(step, text, cls = 'token') {
    const el = document.getElementById(`log-${step}`);
    if (!el) return;
    const span = document.createElement('span');
    span.className = `log-${cls}`;
    span.textContent = text;
    el.appendChild(span);
    el.scrollTop = el.scrollHeight;
  }

  function setColStatus(step, state, label = null) {
    const el = document.getElementById(`status-${step}`);
    if (el) {
      el.className = `col-status ${state}`;
      el.textContent = label || state;
    }
    const dot = document.getElementById(`dot-${step}`);
    if (dot) {
      dot.className = `status-dot ${state}`;
    }
  }

  function setColFooter(step, text) {
    const el = document.getElementById(`footer-${step}`);
    if (el) {
      el.className = `col-footer done`;
      const base = el.textContent.split('—')[0].trim();
      el.textContent = `${base} — ${text}`;
    }
  }

  function setGlobalStatus(state) {
    const dot = document.getElementById('global-dot');
    dot.className = `status-dot ${state === 'running' ? 'running' : state === 'done' ? 'done' : state === 'error' ? 'error' : ''}`;
  }

  function clearAll() {
    ['supervisor','ocr','json','browser'].forEach(s => {
      const el = document.getElementById(`log-${s}`);
      if (el) el.innerHTML = '';
      setColStatus(s, '');
    });
    document.getElementById('run-status').textContent = '';
    document.getElementById('metrics-left').textContent = 'Ready';
    setGlobalStatus('idle');
  }

  // ── Keyboard shortcuts ──
  document.addEventListener('keydown', (e) => {
    if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') return;
    if (e.key === 'r' || e.key === 'R') runPipeline();
    if (e.key === 's' || e.key === 'S') stopPipeline();
    if (e.key === 'd' || e.key === 'D') clearAll();
  });
```
